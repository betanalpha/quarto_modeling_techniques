\begin{tikzpicture}[scale=1]

    % Component One
    % 0, 2, 3, 6, 10, 11, 14, 15
    \foreach \ni/\nf in {0/2, 0/10, 2/6, 3/6, 3/10, 3/14, 11/14, 14/15} {

           ({ \R * cos(\thetai) }, { \R * sin(\thetai)} )
        -- ({ \R * cos(\thetaf) }, { \R * sin(\thetaf)} );
    }
    
    % Component Two
    % 1, 4, 5, 7, 8, 9, 12, 13

           ({ \R * cos(\thetai) }, { \R * sin(\thetai)} )
        -- ({ \R * cos(\thetaf) }, { \R * sin(\thetaf)} );
    }
    
    \foreach \n in {0, 1, ..., 15} {

      \fill[dark]  ({ \R * cos(\theta) }, { \R * sin(\theta)} ) circle (0.075);
